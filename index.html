    // --- БАЗА БИЛАН ИШЛАШ ---
    async function loadAds() {
        const { data, error } = await supabase.from('ads').select('*').order('created_at', { ascending: false });
        const listDiv = document.getElementById('list');
        if (error) { listDiv.innerHTML = "Ошибка базы данных"; return; }
       
        listDiv.innerHTML = data.map(ad => `
            <div class="card">
                <span class="city-tag">${ad.city}</span>
                <h4 style="margin:10px 0;">${ad.title}</h4>
                <p style="color:#666; font-size:14px;">${ad.description}</p>
                <div style="display:flex; justify-content:space-between; align-items:center;">
                    <b style="color:var(--g);">${ad.salary} сомони</b>
                    <button class="btn btn-primary" style="width:auto; padding:8px 15px;" onclick="alert('Связаться: ${ad.phone}')">Позвонить</button>
                </div>
            </div>
        `).join('');
    }

    async function saveAd() {
        const title = document.getElementById('t').value;
        const description = document.getElementById('d').value;
        const salary = document.getElementById('p').value;
        const city = document.getElementById('c').value;
        const phone = document.getElementById('ph').value;

        if (!title || !phone) { alert("Заполните все поля!"); return; }

        const { error } = await supabase.from('ads').insert([{ title, description, salary, city, phone }]);
       
        if (error) {
            alert("Ошибка сохранения: " + error.message);
        } else {
            // SMS ЮБОРИШ (Traccar)
            fetch(smsUrl, {
                method: "POST",
                headers: { "Authorization": smsToken, "Content-Type": "application/json" },
                body: JSON.stringify({ to: phone, message: "Ваше объявление на JobJobs опубликовано!" })
            });
            alert("Объявление успешно добавлено!");
            showPage('home');
        }
    }
